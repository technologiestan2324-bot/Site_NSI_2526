import heapq

def dijkstra(graphe, depart, arrivee):
    # Initialisation
    distances = {sommet: float('inf') for sommet in graphe}
    distances[depart] = 0
    predecesseurs = {sommet: None for sommet in graphe}
    
    # File de priorit√© : (distance, sommet)
    file = [(0, depart)]
    visites = set()
    
    while file:

        if sommet_actuel in visites:
            continue
            
        visites.add(sommet_actuel)
        
        # Si on a atteint l'arriv√©e
        if sommet_actuel == arrivee:
            break
        
        # Examiner les voisins

            # Si on trouve un chemin plus court
            if distance < distances[voisin]:
                distances[voisin] = distance
                predecesseurs[voisin] = sommet_actuel
                heapq.heappush(file, (distance, voisin))
    
    # Reconstituer le chemin
    chemin = []
    sommet = arrivee

    chemin.reverse()
    
    return distances[arrivee], chemin
